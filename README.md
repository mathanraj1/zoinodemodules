## Install

```bash
npm install --save zoinodemodules
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'zoinodemodules'
import 'zoinodemodules/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
